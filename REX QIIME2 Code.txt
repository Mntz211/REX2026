## Tree 
# Generate a tree for phylogenetic diversity analyses
qiime phylogeny align-to-tree-mafft-fasttree \
  --i-sequences bankvole_repseqs.qza \
  --o-alignment aligned-rep-seqs.qza \
  --o-masked-alignment masked-aligned-rep-seqs.qza \
  --o-tree unrooted-tree.qza \
  --o-rooted-tree vole-tree.qza

qiime tools export \
--input-path ../REX/vole-tree.qza \
--output-path vole-rooted-tree_export

#Send tree.nwk to local computer
scp root@10.19.139.127:/data/REX/Vole/vole-rooted-tree_export/tree.nwk .
